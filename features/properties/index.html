Invoking properties from QML.

public class NetObject
{
    public NetObject()
    {
        Prop = "hello!";
    }

    public string Prop { get; set; }
}
import YourApp 2.1

YourNetObject {
    id: o
    Component.onCompleted: {
        // Outputs "hello!"
        console.log(o.Prop)
    }
}
